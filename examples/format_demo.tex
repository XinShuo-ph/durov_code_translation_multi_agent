\documentclass[12pt,a4paper]{article}
\usepackage{xeCJK}
\usepackage{fontspec}
\usepackage{xcolor}
\usepackage{geometry}
\usepackage{setspace}

% Page setup
\geometry{margin=2.5cm}
\setstretch{1.3}

% Font setup for multiple languages
\setmainfont{DejaVu Serif}
\setCJKmainfont{Noto Sans CJK SC}

% Set up Chinese and Japanese fonts
\setCJKfamilyfont{zhsong}{Noto Serif CJK SC}
\setCJKfamilyfont{jpgothic}{Noto Sans CJK JP}

% Define colors for each language
\definecolor{russian}{RGB}{0,0,0}        % Black
\definecolor{english}{RGB}{0,102,204}    % Blue
\definecolor{chinese}{RGB}{204,0,0}      % Red
\definecolor{japanese}{RGB}{0,153,76}    % Green

% Commands for each language with color
\newcommand{\ru}[1]{{\color{russian}#1}}
\newcommand{\en}[1]{{\color{english}#1}}
\newcommand{\zh}[1]{{\color{chinese}\CJKfamily{zhsong}#1}}
\newcommand{\ja}[1]{{\color{japanese}\CJKfamily{jpgothic}#1}}
